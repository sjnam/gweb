% gwebmac.tex -- plain-TeX macros for documents produced by gweave.
% This is the GWEB analogue of CWEB's cwebmac.tex; it is independent macro code
% written to give gweave output the same page layout that cweave produces.
% Process a woven file with a plain-TeX engine, e.g. pdftex foo.tex.

% ---- fonts ------------------------------------------------------------------
\font\eightrm=cmr8                       % small roman for notes, headers, index
\font\titlefont=cmr7 scaled\magstep4     % big title on the contents page
\let\mainfont=\tenrm

% \.{text} sets verbatim/typewriter text in prose (as in cwebmac). Inside it the
% escaped specials \\ \{ \} \& \# \% \$ \^ \_ \~ print as the literal characters,
% so e.g. \.{@\^} prints the control code @^. Used for file names, command names,
% and control codes in commentary and section titles. (Plain TeX's \. is a dot
% accent; we override it -- gweave documents need \tt strings, not accents.)
% This is a plain \def (not e-TeX's \protected, so bare `tex' works for the DVI
% route): the running-head and contents machinery below is written to carry the
% title without ever expanding \. (see \Gsetmark and \Grunhead).
\def\.#1{{\tt
  \def\\{\char92 }\def\{{\char123 }\def\}{\char125 }%
  \def\&{\char38 }\def\^{\char94 }\def\_{\char95 }\def\~{\char126 }%
  \def\${\char36 }\def\#{\char35 }\def\%{\char37 }%
  #1}}

% ---- page layout ------------------------------------------------------------
\newdimen\Gpagewidth  \Gpagewidth=6.5in
\newdimen\Gpageheight \Gpageheight=8.7in
\newdimen\Gcolwidth   \Gcolwidth=.5\Gpagewidth \advance\Gcolwidth by-10pt
\hsize=\Gpagewidth
\vsize=\Gpageheight
\parindent=1em
\parskip=0pt
\hbadness=10000
\tolerance=2000
\newdimen\Gindentunit \Gindentunit=1.4em % one indentation step in displayed code
\newcount\Gtmpcount \newdimen\Gtmpdimen  % scratch registers (avoid e-TeX exprs)
\newskip\Gsecskip     \Gsecskip=12pt plus 4pt minus 3pt % space between sections
\newcount\Gsecpagedepth \Gsecpagedepth=1 % starred sections this deep start a page

% ---- running headers --------------------------------------------------------
% The document title is the upper-cased job name; the current group title is set
% by each starred section. Each section deposits a \mark of the form
% {section-number}{group-title}; the headline reads it back through \topmark.
% Upper-case the (expanded) job name for the running header and contents title.
% \title is shown in the running header and on the contents page; it defaults to
% the upper-cased job name and may be redefined in limbo.
\edef\title{\jobname}
\def\Gtitleaux#1{\uppercase{\gdef\title{#1}}}
\expandafter\Gtitleaux\expandafter{\title}
\def\Gtitle{\title}
\let\GgroupUP\Gtitle
\def\Ggroup{\GgroupUP}
% The mark carries the group title without expanding it (\noexpand\Ggroup), so a
% \. in the title survives into \topmark for typesetting in the header.
\def\Gsetmark#1{\mark{{#1}{\noexpand\Ggroup}}}
\def\Gtakeone#1#2{#1}
\def\Gtaketwo#1#2{#2}
\def\Gempty{}

\newif\iftitle % true only while shipping the contents page (suppresses the header)
\def\contentspagenumber{0} % page number of the contents page; redefine in limbo
\def\Gsechead{\S\thinspace\expandafter\Gtakeone\Gtm} % left/right end: section number
\def\Ggrouphead{\expandafter\Gtaketwo\Gtm}   % middle: current group title
% \Grunhead builds the header as one \Gpagewidth-wide box (used both by the
% normal \headline and by the index's own output routine).
\def\Grunhead{%
  % Capture the mark tokens without expanding them (a \. in the title must not be
  % expanded here); \expandafter expands \topmark exactly once into \Gtm.
  \expandafter\def\expandafter\Gtm\expandafter{\topmark}%
  \ifx\Gtm\Gempty\expandafter\def\expandafter\Gtm\expandafter{\firstmark}\fi
  \ifx\Gtm\Gempty\def\Gtm{{}{\Gtitle}}\fi
  \ifodd\pageno
    \hbox to\Gpagewidth{\mainfont\Gsechead\enspace{\eightrm\Gtitle}\hfil
      {\eightrm\Ggrouphead}\enspace\folio}%
  \else
    \hbox to\Gpagewidth{\mainfont\folio\enspace{\eightrm\Ggrouphead}\hfil
      {\eightrm\Gtitle}\enspace\Gsechead}%
  \fi}
\headline={\iftitle\hfil\else\Grunhead\fi}
\footline={\hfil}

% ---- hyperlinks (active in pdfTeX PDF mode; a no-op for dvi) ------------------
% Each section deposits a numbered destination; section numbers shown as
% references, in the index, in cross-reference notes, and in the contents are
% links to those destinations, drawn in blue like cweave.
\newif\ifGpdf
\ifx\pdfoutput\undefined\else\ifnum\pdfoutput>0 \Gpdftrue\fi\fi
\def\Gdest#1{\ifGpdf\pdfdest num #1 fith\relax\fi}
\def\Glink#1#2{\ifGpdf
    \pdfstartlink attr{/Border[0 0 0]}goto num #1\relax
    \pdfliteral{0 0 1 rg}#2\pdfliteral{0 g}\pdfendlink
  \else #2\fi}
\def\Gs#1{\Glink{#1}{#1}}                        % a linked section number
\def\GsD#1{\Glink{#1}{$\underline{\hbox{#1}}$}}  % linked + underlined (definition)
% A PDF outline (bookmark) entry: #1 section number, #2 number of direct
% children (its sign opens/closes the entry), #3 the plain-text title.
\def\Gbookmark#1#2#3{\ifGpdf\pdfoutline goto num #1 count #2 {\pdfescapestring{#3}}\fi}
% \Gmakeheadline places \Grunhead like plain TeX's \makeheadline, for use inside
% the index's custom \shipout (where the plain \headline machinery is bypassed).
\def\Gmakeheadline{\vbox to 0pt{\vskip-22.5pt
  \hbox to\Gpagewidth{\vbox to8.5pt{}\Grunhead}\vss}\nointerlineskip}

% A starred section records itself in \jobname.toc as it is shipped, so \Gcon can
% build the contents in one pass; the page number is captured by the deferred
% \write, the title is stashed in a control sequence.
\newwrite\Gtoc
\immediate\openout\Gtoc=\jobname.toc

% ---- section openers --------------------------------------------------------
% \GM{n} : an ordinary (unstarred) section, run-in bold number.
\def\GM#1{\par\penalty-100\vskip\Gsecskip
  \Gsetmark{#1}\noindent\Gdest{#1}{\bf #1.\quad}\rightskip=0pt\ignorespaces}

% \GN{depth}{n}{title} : a starred section. Major sections (depth < secpagedepth)
% begin a new page; the first one does not, so it follows any limbo banner.
\newif\ifGfirst \Gfirsttrue
\def\GN#1#2#3{\par
  % \gdef (not \xdef) so a \. in the title is stored unexpanded; \uppercase only
  % upper-cases the letter tokens, leaving control sequences intact.
  {\uppercase{\gdef\GgroupUP{#3}}}\global\let\Ggroup\GgroupUP
  \expandafter\gdef\csname Gtoctitle\number#2\endcsname{#3}%
  \Gsetmark{#2}% set the mark before any page break, so the new page's \topmark
  \ifGfirst \Gfirstfalse \vskip 6pt
  \else \ifnum#1<\Gsecpagedepth \vfil\eject
        \else \penalty-200\vfil\penalty-200\vfilneg\vskip\Gsecskip \fi
  \fi
  \write\Gtoc{\string\GTline{#1}{#2}{\the\pageno}}%
  \noindent\Gdest{#2}{\bf #2.\quad #3.\quad}\rightskip=0pt\ignorespaces}

% ---- displayed Go code ------------------------------------------------------
\newdimen\Gcodeindent \Gcodeindent=\Gindentunit % displayed code is indented this
                                                % far (about one tab) from the margin
\def\GB{\par\nobreak\vskip 3pt\begingroup
  \rightskip=0pt plus 1fil \parfillskip=0pt plus 1fil
  \hfuzz=2pt \emergencystretch=3em \parindent=0pt}
\def\GE{\par\endgroup\vskip 3pt}
% One displayed code line: #1 indentation steps, #2 the body (math chunks $...$
% joined by \GS). The whole display is offset by \Gcodeindent from the margin;
% long lines wrap at a \GS, continuation indented one step deeper.
\def\GL#1#2{\par
  \Gtmpcount=#1 \Gtmpdimen=\Gtmpcount\Gindentunit \advance\Gtmpdimen by\Gcodeindent
  \hangindent=\Gtmpdimen \advance\hangindent by\Gindentunit \hangafter=1
  \noindent\kern\Gtmpdimen \strut #2}
\def\GS{\penalty0\hskip .5em plus .3em minus .15em\relax} % breakable code space
\def\GSO{\penalty0\hskip 0pt\relax}                       % @| optional break
\def\GBL{\par\smallskip}                                  % @# blank line in code
\def\GBK{\par\vskip 2.5pt}    % a blank source line: a little air between groups

% ---- code token fonts (called inside the math of \GL) -----------------------
\def\GID#1{\hbox{\it #1\/}}      % identifier
\def\GKW#1{\hbox{\bf #1}}        % reserved word or predeclared type
\def\GST#1{\hbox{\tt #1}}        % string, rune, or verbatim text
\def\GCM#1{\hbox{\rm #1}}        % comment
\def\GNU#1{\hbox{\rm #1}}        % numeric literal

% ---- named-section references and headlines ---------------------------------
\def\Gangle#1#2{$\langle\,$#1\hbox{\eightrm\kern.4em\Gs{#2}}$\,\rangle$}
\def\GX#1#2{\hbox{\Gangle{#2}{#1}}}        % inline reference in code
\def\GD#1#2{\par\vskip 3pt\noindent\hbox{\Gangle{#2}{#1}${}\equiv{}$}\hskip 3pt}
\def\GDp#1#2{\par\vskip 3pt\noindent
  \hbox{\Gangle{#2}{#1}${}\mathrel{+}\equiv{}$}\hskip 3pt}

% ---- cross-reference notes printed under a definition -----------------------
\def\Gnote#1{\par\penalty-1\vskip 2pt\noindent{\eightrm #1.}\par}
\def\GU#1{\Gnote{This code is used in section #1}}
\def\GUs#1{\Gnote{This code is used in sections #1}}
\def\GA#1{\Gnote{This code is also defined in section #1}}
\def\GAs#1{\Gnote{This code is also defined in sections #1}}

% ---- index: title-less, set in two columns ----------------------------------
% The index is set in two columns. \Ginx stashes the preceding material (the
% user's `@* Index.' heading and prose) in \Gsbox, then narrows \hsize to one
% column and installs \Gcoloutput: it fills the left column to the column height,
% saves it, fills the right column, and ships a page carrying the stashed
% material above the two columns. \Gfin flushes the last column.
\newbox\Gsbox \newbox\Glbox
\def\Glr{L}
\def\Gcoloutput{%
  \if L\Glr
    \global\setbox\Glbox=\box255 \gdef\Glr{R}%
  \else
    \shipout\vbox{\Gmakeheadline
      \vbox to\Gpageheight{\boxmaxdepth=\maxdimen
        \box\Gsbox\vss
        \hbox to\Gpagewidth{\box\Glbox\hfil\box255}}}%
    \global\advance\pageno by1
    \global\setbox\Gsbox=\vbox{}\global\vsize=\Gpageheight \gdef\Glr{L}%
  \fi}
\def\Ginx{\par
  \vskip 9pt plus 2pt           % gap between the Index heading/prose and entries
  \output={\global\setbox\Gsbox=\box255}\eject % stash heading+prose, hold page
  \setbox\Gsbox=\vbox{\unvbox\Gsbox}%
  \global\vsize=\Gpageheight
  \global\advance\vsize by-\ht\Gsbox \global\advance\vsize by-\dp\Gsbox
  \global\hsize=\Gcolwidth
  \global\output={\Gcoloutput}\gdef\Glr{L}%
  \begingroup
  \baselineskip=12pt \parindent=0pt \parskip=1pt plus .5pt \rm
  \rightskip=0pt plus 2.5em \tolerance=10000 \hyphenpenalty=10000}
\def\GII#1#2{\noindent\hangindent2em #1:\kern.5em #2.\par} % one index entry
\def\GIR#1{{\rm #1}}                          % @^...@>  roman index entry
\def\GIT#1{{\tt #1}}                          % @....@>  typewriter index entry
\def\9#1{{\rm #1}}                            % default for @:...@> entries
\def\GIC#1{\9{#1}}

% ---- list of section names (a `Names of the Sections' page) -----------------
\def\Gfin{\par\vfil\eject       % seal/ship the current index column
  \if L\Glr\else\null\vfil\eject\fi % a left column is pending: ship its page
  \endgroup                     % close the index paragraph group
  \global\output={\plainoutput}% back to ordinary single-column pages
  \global\hsize=\Gpagewidth \global\vsize=\Gpageheight
  \vfil\eject
  \global\def\Gsechead{}% no section number in the header here
  \global\def\Ggrouphead{NAMES OF THE SECTIONS}%
  \begingroup
  \parindent=0pt \parskip=2pt plus 1pt \rightskip=0pt plus 2em \rm}
\def\GNS#1#2#3{\noindent\hangindent2em \Gangle{#1}{#2}%
  \ifx\Gempty#3\else\quad{\eightrm#3}\fi\par}

% ---- contents page: centered title, Section/Page columns, dot leaders --------
\def\GTline#1#2#3{\line{\Gconsetup{#1}%
  \csname Gtoctitle\number#2\endcsname\ %
  \rm\leaders\hbox to .5em{\hss.\hss}\hfil\ \Gs{#2}\hbox to 3em{\hss#3}}}
\def\Gconsetup#1{\ifcase#1 \or\hskip2em\or\hskip4em\or\hskip6em\else\hskip8em\fi}
\def\Gcon{\par\endgroup\vfil\eject % finish the section names
  \immediate\closeout\Gtoc
  \titletrue \pageno=\contentspagenumber
  \parindent=0pt
  \centerline{\titlefont\Gtitle}\vskip .7in \vfill
  \line{\hfil{\eightrm Section}\hbox to 3em{\hss\eightrm Page}}%
  \begingroup \rm
    \input \jobname.toc
  \endgroup
  \vfill\end}
