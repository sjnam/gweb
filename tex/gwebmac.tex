% gwebmac.tex -- plain-TeX macros for documents produced by gweave.
% This is the GWEB analogue of CWEB's cwebmac.tex; it is independent macro code
% written to give gweave output the same page layout that cweave produces.
% Process a woven file with a plain-TeX engine, e.g. pdftex foo.tex.

% ---- fonts ------------------------------------------------------------------
% The font set mirrors cwebmac.tex so woven output matches cweave's typefaces.
\font\eightrm=cmr8                       % small roman for notes, headers, index
\font\ninerm=cmr9 \let\mc=\ninerm        % medium caps (\CEE/, \GO/, ...)
\let\sc=\eightrm                         % smallish caps
\font\titlefont=cmr7 scaled\magstep4     % big title on the contents page
\font\ttitlefont=cmtt10 scaled\magstep2  % typewriter title (for use in limbo)
\font\tentex=cmtex10                      % TeX extended set, for strings (as cweb)
\fontdimen7\tentex=0pt                    % no extra space after sentences
\let\mainfont=\tenrm
\let\cmntfont=\tenrm                      % font for comments (cf. cweb's \cmntfont)

% Language and prose helpers borrowed from cwebmac, for use in commentary; each
% is invoked with a trailing slash, e.g. ``written in \GO/''.
\def\CEE/{{\mc C\spacefactor1000}}
\def\GO/{{\mc Go\spacefactor1000}}
\def\UNIX/{{\mc U\kern-.05emNIX\spacefactor1000}}
\def\TEX/{\TeX}
% \\{id} sets an identifier in italics in prose, as in cweb. (The GWEB idiom is
% to write inline code as |id|; this is provided for porting cweb commentary.
% cweb's \& and \| are deliberately not redefined: \& is used here as a literal
% ampersand by the comment/section-name escaper, and gweave rewrites \| to a bar.)
\def\\#1{\leavevmode\hbox{\it#1\/\kern.05em}}
% A thin space that works in either math or text mode (cwebmac's \,).
\def\,{\relax\ifmmode\mskip\thinmuskip\else\thinspace\fi}

% \.{text} sets verbatim/typewriter text in prose (as in cwebmac). Inside it the
% escaped specials \\ \{ \} \& \# \% \$ \^ \_ \~ print as the literal characters,
% so e.g. \.{@\^} prints the control code @^. Used for file names, command names,
% and control codes in commentary and section titles. (Plain TeX's \. is a dot
% accent; we override it -- gweave documents need \tt strings, not accents.)
% This is a plain \def (not e-TeX's \protected, so bare `tex' works for the DVI
% route): the running-head and contents machinery below is written to carry the
% title without ever expanding \. (see \Gsetmark and \Grunhead).
\def\.#1{{\tentex
  \def\\{\char92 }\def\{{\char123 }\def\}{\char125 }%
  \def\&{\char38 }\def\^{\char94 }\def\_{\char95 }\def\~{\char126 }%
  \def\${\char36 }\def\#{\char35 }\def\%{\char37 }%
  #1}}

% ---- page layout ------------------------------------------------------------
\newdimen\Gpagewidth  \Gpagewidth=6.5in
\newdimen\Gpageheight \Gpageheight=8.7in
\newdimen\Gcolwidth   \Gcolwidth=.5\Gpagewidth \advance\Gcolwidth by-10pt
\hsize=\Gpagewidth
\vsize=\Gpageheight
\parindent=1em
\parskip=0pt
\hbadness=10000
\tolerance=2000
\newdimen\Gindentunit \Gindentunit=1.4em % one indentation step in displayed code
\newcount\Gtmpcount \newdimen\Gtmpdimen  % scratch registers (avoid e-TeX exprs)
\newskip\Gsecskip     \Gsecskip=12pt plus 4pt minus 3pt % space between sections
\newcount\Gcodepenalty \Gcodepenalty=9999 % discourage page breaks inside code
                                          % (high but finite, so >1-page code can still break)
\newdimen\Gcommentkern \Gcommentkern=-1.3pt % tighten the "//" comment marker
\newcount\Gsecpagedepth \Gsecpagedepth=1 % starred sections this deep start a page
\newmuskip\Gthinmuskip \Gthinmuskip=1.5mu % gweave's thin space before "(" and "["
\def\Gthin{\mskip\Gthinmuskip}          % (narrower than \, which is 3mu)

% ---- running headers --------------------------------------------------------
% The document title is the upper-cased job name; the current group title is set
% by each starred section. Each section deposits a \mark of the form
% {section-number}{group-title}; the headline reads it back through \topmark.
% Upper-case the (expanded) job name for the running header and contents title.
% \title is shown, as written, in the running header and on the contents page. It
% defaults to the upper-cased job name -- so a web with no \def\title still gets
% an all-caps header, as in cweb -- but a \def\title{...} in limbo is shown
% verbatim, mixed case and all: cweb likewise upper-cases only the default
% job-name title, never one the author supplies.
\edef\title{\jobname}
\def\Gtitleaux#1{\uppercase{\gdef\title{#1}}}
\expandafter\Gtitleaux\expandafter{\title}
\def\Gtitle{\title}
% The current group (chapter) title, upper-cased, is kept in a token register so
% each \mark can capture a *snapshot* of it (via \the) rather than a reference to
% a macro -- otherwise every page would show whatever group is current at shipout
% time, i.e. the next chapter. \the inserts the tokens without expanding them, so
% a \. in the title survives into \topmark for typesetting in the header.
\newtoks\Ggrouptoks
\def\Gsetmark#1{\mark{{#1}{\the\Ggrouptoks}}}
\def\Gtakeone#1#2{#1}
\def\Gtaketwo#1#2{#2}
\def\Gempty{}

\newif\iftitle % true only while shipping the contents page (suppresses the header)
\def\contentspagenumber{0} % page number of the contents page; redefine in limbo
\def\Gsechead{\S\thinspace\expandafter\Gtakeone\Gtm} % left/right end: section number
\def\Ggrouphead{\expandafter\Gtaketwo\Gtm}   % middle: current group title
% Font for the running-head title and group title (small roman, as in cweb).
% A localization file may rebind it -- e.g. kotexgweb.tex pairs it with a sans
% hangul font so Korean in the header matches the upper-cased Latin.
\def\Gheadfont{\eightrm}
% Font for the small print: cross-reference notes, the contents columns, and the
% usage note in the section-name list. A localization file may rebind it -- e.g.
% kotexgweb.tex pairs it with an 8pt hangul font so Korean there is small too.
\def\Gsmallfont{\eightrm}
% \Grunhead builds the header as one \Gpagewidth-wide box (used both by the
% normal \headline and by the index's own output routine).
\def\Grunhead{%
  % Capture the mark tokens without expanding them (a \. in the title must not be
  % expanded here); \expandafter expands \topmark exactly once into \Gtm.
  \expandafter\def\expandafter\Gtm\expandafter{\topmark}%
  \ifx\Gtm\Gempty\expandafter\def\expandafter\Gtm\expandafter{\firstmark}\fi
  \ifx\Gtm\Gempty\def\Gtm{{}{\Gtitle}}\fi
  \ifodd\pageno
    \hbox to\Gpagewidth{\mainfont\Gsechead\enspace{\Gheadfont\Gtitle}\hfil
      {\Gheadfont\Ggrouphead}\enspace\folio}%
  \else
    \hbox to\Gpagewidth{\mainfont\folio\enspace{\Gheadfont\Ggrouphead}\hfil
      {\Gheadfont\Gtitle}\enspace\Gsechead}%
  \fi}
\headline={\iftitle\hfil\else\Grunhead\fi}
\footline={\hfil}

% ---- hyperlinks --------------------------------------------------------------
% Each section deposits a numbered destination; section numbers shown as
% references, in the index, in cross-reference notes, and in the contents link
% to those destinations, drawn in blue like cweave. Two back ends are supported:
%   * pdfTeX/luaTeX primitives, when producing PDF directly (\pdfoutput>0);
%   * \special{pdf:...} commands for the  tex -> dvipdfmx  route. As in cweb,
%     that route is requested by defining \pdf, e.g.
%         tex "\let\pdf+ \input foo.tex"   then   dvipdfmx foo.dvi
% With a plain dvi engine and no \pdf, the links and bookmarks are omitted.
% For the dvipdfmx route, destinations are named with roman numerals (as in
% cwebmac); links and bookmarks reference those names, so they work even though
% gweave emits all the bookmarks together at the end of the document.
\newif\ifGpdf      % native pdfTeX/luaTeX PDF output
\newif\ifGdvipdf   % dvipdfmx \special{pdf:...} route
\ifx\pdfoutput\undefined\else\ifnum\pdfoutput>0 \Gpdftrue\fi\fi
\ifGpdf\else\ifx\pdf\undefined\else\Gdvipdftrue\fi\fi
\newbox\Glinkbox
\def\Gdest#1{%
  \ifGpdf \pdfdest num #1 fith\relax
  \else\ifGdvipdf
    \smash{\raise\baselineskip\hbox to0pt{%
      \special{pdf: dest (\romannumeral#1) [ @thispage /FitH @ypos ]}\hss}}%
  \fi\fi}
% The destination for the `Names of the sections' page. It is emitted only when
% that list is (\ifGsecs), so \nosecs leaves no stray whatsit --- and hence no
% blank page --- on the sheet the index ejects into.
\def\Gsecdest#1{\ifGsecs\Gdest{#1}\fi}
\def\Glink#1#2{%
  \ifGpdf
    \pdfstartlink attr{/Border[0 0 0]}goto num #1\relax
    \pdfliteral{0 0 1 rg}#2\pdfliteral{0 g}\pdfendlink
  \else\ifGdvipdf
    \setbox\Glinkbox=\hbox{\special{pdf: bc [ 0 0 1 ]}#2\special{pdf: ec}}%
    \special{pdf: ann width \the\wd\Glinkbox\space height \the\ht\Glinkbox\space
       depth \the\dp\Glinkbox\space << /Subtype /Link /Border [ 0 0 0 ]
       /A << /S /GoTo /D (\romannumeral#1) >> >>}\box\Glinkbox\relax
  \else #2\fi\fi}
\def\Gs#1{\Glink{#1}{#1}}                        % a linked section number
\def\GsD#1{\Glink{#1}{$\underline{\hbox{#1}}$}}  % linked + underlined (definition)
% A PDF outline (bookmark) entry: #1 group depth, #2 section number, #3 number of
% direct children (pdfTeX's open/close model), #4 the plain-text title. The
% dvipdfmx route nests by depth and links to the named destination.
\def\Gbookmark#1#2#3#4{%
  \ifGpdf \pdfoutline goto num #2 count #3 {\pdfescapestring{#4}}%
  \else\ifGdvipdf
    \special{pdf: outline #1 << /Title (#4)
       /A << /S /GoTo /D (\romannumeral#2) >> >>}%
  \fi\fi}
% \Gmakeheadline places \Grunhead like plain TeX's \makeheadline, for use inside
% the index's custom \shipout (where the plain \headline machinery is bypassed).
\def\Gmakeheadline{\vbox to 0pt{\vskip-22.5pt
  \hbox to\Gpagewidth{\vbox to8.5pt{}\Grunhead}\vss}\nointerlineskip}

% A starred section records itself in \contentsfile as it is shipped, so \Gcon
% can build the contents in one pass; the page number is captured by the deferred
% \write, the title is stashed in a control sequence. \contentsfile (the file the
% data goes to) and \readcontents (how \Gcon reads it back) are redefinable in
% limbo, as in cwebmac: to split a long document into separately-typeset pieces
% and gather their contents in the last one, give each piece its own
% \contentsfile and let the last \readcontents \input them all. The file is
% opened lazily, by \Gopentoc at the first starred section, so that a limbo
% redefinition of \contentsfile is already in force when it opens.
\def\contentsfile{\jobname.toc}
\def\readcontents{\input \contentsfile }
\newwrite\Gtoc
\newif\ifGtocopen
\def\Gopentoc{\ifGtocopen\else\Gtocopentrue\immediate\openout\Gtoc=\contentsfile \fi}

% ---- section openers --------------------------------------------------------
% \GM{n} : an ordinary (unstarred) section, run-in bold number. The
% \vfil\penalty-100\vfilneg makes the section boundary the preferred page break
% (the \vfil fills the page there, so a short page costs nothing): as in cweave,
% TeX breaks between sections rather than inside one, so a whole section -- its
% commentary and its code -- stays together on a page when it fits.
\def\GM#1{\par\Gemitdate\vfil\penalty-100\vfilneg\vskip\Gsecskip
  \Gsetmark{#1}\noindent\Gdest{#1}{\bf #1.\quad}\rightskip=0pt\ignorespaces}

% \GN{depth}{n}{title} : a starred section. Major sections (depth < secpagedepth)
% begin a new page; the first one does not, so it follows any limbo banner.
\newif\ifGfirst \Gfirsttrue
\def\GN#1#2#3{\par\Gemitdate
  % Snapshot the upper-cased title into \Ggrouptoks; \uppercase only upper-cases
  % the letter tokens, leaving control sequences (\.) intact for the header.
  {\uppercase{\global\Ggrouptoks={#3}}}%
  \Gsetmark{#2}% set the mark before any page break, so the new page's \topmark
  \ifGfirst \Gfirstfalse \vskip 6pt
  \else \ifnum#1<\Gsecpagedepth \vfil\eject
        \else \penalty-200\vfil\penalty-200\vfilneg\vskip\Gsecskip \fi
  \fi
  \Gopentoc
  % The title (#3) is written unexpanded so it survives to the contents run
  % intact, and the page number is captured by the deferred \write after the break.
  \write\Gtoc{\string\GTline{#1}{#2}{\unexpanded{#3}}{\the\pageno}}%
  \noindent\Gdest{#2}{\bf #2.\quad #3.\quad}\rightskip=0pt\ignorespaces}

% ---- \datethis: timestamp the listing just above section 1 ------------------
% cweave's \datethis (said in limbo) prints the date and time, in \eightrm (its
% \sc), on a left-flush line above the first section. \today and \Ghours compute
% them; \Gdatefont and \Gdateat -- the font and the date/time separator -- are
% overridable, so the Korean backend can localize the wording and the font size.
\def\today{\ifcase\month\or
  January\or February\or March\or April\or May\or June\or
  July\or August\or September\or October\or November\or December\fi
  \space\number\day, \number\year}
\newcount\Gtwodigits
\def\Ghours{\Gtwodigits=\time \divide\Gtwodigits by60 \Gprinttwo
  \multiply\Gtwodigits by-60 \advance\Gtwodigits by\time :\Gprinttwo}
\def\Ggobbleone1{}
\def\Gprinttwo{\advance\Gtwodigits by100
  \expandafter\Ggobbleone\number\Gtwodigits \advance\Gtwodigits by-100 }
\def\Gdatefont{\eightrm}  % cweb sets this line in \sc, i.e. \eightrm (8pt)
\def\Gdateat{\ at\ }      % between the date and the time
\def\Gdateline{\leftline{\Gdatefont\today\Gdateat\Ghours}\bigskip}
\let\Gdatehook\empty
\def\datethis{\let\Gdatehook\Gdateline}
\def\Gemitdate{\Gdatehook\global\let\Gdatehook\empty} % fires once, at section 1

% ---- displayed Go code ------------------------------------------------------
\newdimen\Gcodeindent \Gcodeindent=\Gindentunit % displayed code is indented this
                                                % far (about one tab) from the margin
% \penalty\Gcodepenalty before the code keeps it joined to its commentary (the
% section breaks as a unit); like the penalty inside \GL it is high but finite,
% so a section too tall for a page can still break, preferring the commentary.
\def\GB{\par\penalty\Gcodepenalty\vskip 3pt\begingroup
  \rightskip=0pt plus 1fil \parfillskip=0pt plus 1fil
  \hfuzz=2pt \emergencystretch=3em \parindent=0pt}
% \GBr: run-in code start for a code-only section with no commentary -- no \par,
% so the first code line (set with \GLr) follows the section number; used by
% cweave-style code-only sections that have no name and no commentary.
\def\GBr{\penalty\Gcodepenalty\begingroup
  \rightskip=0pt plus 1fil \parfillskip=0pt plus 1fil
  \hfuzz=2pt \emergencystretch=3em \parindent=0pt}
\def\GE{\par\endgroup\vskip 3pt}
% One displayed code line: #1 indentation steps, #2 the body (math chunks $...$
% joined by \GS). The whole display is offset by \Gcodeindent from the margin;
% long lines wrap at a \GS, continuation indented one step deeper.
% \penalty\Gcodepenalty before each line discourages (but, being finite, does not
% forbid) a page break between consecutive code lines, so a code part prefers to
% stay together; a code part longer than a page can still break.
\def\GL#1#2{\par\penalty\Gcodepenalty
  \Gtmpcount=#1 \Gtmpdimen=\Gtmpcount\Gindentunit \advance\Gtmpdimen by\Gcodeindent
  \hangindent=\Gtmpdimen \advance\hangindent by\Gindentunit \hangafter=1
  \noindent\kern\Gtmpdimen \strut #2}
% \GLr: the run-in first code line of a code-only section -- no \par and no
% indent, so it continues on the section-number line; later lines use \GL.
\def\GLr#1#2{\strut #2}
\def\GS{\penalty0\hskip .222em plus .12em minus .08em\relax} % breakable code space (cweb medmuskip)
\def\GSO{\penalty0\hskip 0pt\relax}                       % @| optional break
\def\GBL{\par\penalty\Gcodepenalty\smallskip}             % @# blank line in code
\def\GBK{\par\penalty\Gcodepenalty\vskip 2.5pt} % a blank source line: a little air

% ---- code token fonts (called inside the math of \GL) -----------------------
\def\GID#1{\hbox{\it #1\/}}      % identifier
\def\GKW#1{\hbox{\bf #1}}        % reserved word or predeclared type
\def\GST#1{\hbox{\tentex #1}}    % string, rune, or verbatim text (cweb's \tentex)
\def\GMAC#1{\hbox{\tentex #1}}   % const name, set typewriter like cweb's @d macro
\def\GSP{{\tt\char32 }}          % a visible space inside a string (cweb's \SP)
\def\Gnil{\mathord{\Lambda}}     % Go's nil, shown as cweb shows C's NULL (\Lambda)
\def\GCM#1{\hbox{\cmntfont #1}}  % comment
\def\GNU#1{\hbox{\rm #1}}        % decimal/floating numeric literal
% Non-decimal literals, as in cweb: octal with a raised circle and oldstyle
% italic digits, hex with a superscript "#" in typewriter, binary with a "b".
\def\Goct#1{\hbox{$^\circ$\kern-.1em{\it #1\/}}}
\def\Ghex#1{\hbox{$^{\scriptscriptstyle\#}$\tt #1}}
\def\Gbin#1{\hbox{$^{\scriptscriptstyle b}$\tt #1}}

% ---- named-section references and headlines ---------------------------------
\def\Gangle#1#2{$\langle\,$#1\hbox{\eightrm\kern.4em\Gs{#2}}$\,\rangle$}
\def\GX#1#2{\hbox{\Gangle{#2}{#1}}}        % inline reference in code
\def\GD#1#2{\par\vskip 3pt\noindent\hbox{\Gangle{#2}{#1}${}\equiv{}$}\hskip 3pt}
\def\GDp#1#2{\par\vskip 3pt\noindent
  \hbox{\Gangle{#2}{#1}${}\mathrel{+}\equiv{}$}\hskip 3pt}
% Run-in variants \GDr/\GDpr: used when a section has no commentary, so the
% definition's name and equivalence sign follow the section number on the same

\def\GDr#1#2{\hbox{\Gangle{#2}{#1}${}\equiv{}$}\hskip 3pt}
\def\GDpr#1#2{\hbox{\Gangle{#2}{#1}${}\mathrel{+}\equiv{}$}\hskip 3pt}

% ---- cross-reference notes printed under a definition -----------------------
\def\Gnote#1{\par\penalty-1\vskip 2pt\noindent{\Gsmallfont #1.}\par}
\def\GU#1{\Gnote{This code is used in section #1}}
\def\GUs#1{\Gnote{This code is used in sections #1}}
\def\GA#1{\Gnote{This code is also defined in section #1}}
\def\GAs#1{\Gnote{This code is also defined in sections #1}}
% Other human-language strings: the running head over the section-name list and
% the two column headings on the contents page. Redefine these, and the four
% notes just above, in a localization file (e.g. kotexgweb.tex for Korean) to
% translate gweave's output; gweave itself needs no change.
\def\Gnamesheading{NAMES OF THE SECTIONS}
\def\Goutsecname{Names of the sections} % title of the section-names PDF bookmark
\def\Gsectionword{Section}
\def\Gpageword{Page}
% The usage note printed beside an entry in the section-name list.
\def\GNused#1{Used in section #1}
\def\GNuseds#1{Used in sections #1}

% ---- selectively suppressing the back matter (as in cwebmac) ----------------
% Put any of these in limbo to drop part of the tail. Following cwebmac they are
% cumulative in reach: \noinx drops the index, the `Names of the sections' list,
% and the contents; \nosecs drops the list and the contents; \nocon drops the
% contents. The three flags guard the matching parts of \Ginx/\GII, \Gfin/\GNS,
% and \Gcon below, so every group stays balanced whatever is switched off.
% \noatl is accepted for portability with cwebmac webs but does nothing: GWEB
% has no \.{@l} transliteration (Go source is Unicode), so there is no such note.
\newif\ifGinx  \Ginxtrue   % include the index?
\newif\ifGsecs \Gsecstrue  % include the `Names of the sections' list?
\newif\ifGcon  \Gcontrue   % include the table of contents?
\def\noinx{\Ginxfalse\Gsecsfalse\Gconfalse}
\def\nosecs{\Gsecsfalse\Gconfalse}
\def\nocon{\Gconfalse}
\def\noatl{}

% ---- index: title-less, set in two columns ----------------------------------
% The index is set in two columns. \Ginx stashes the preceding material (the
% user's `@* Index.' heading and prose) in \Gsbox, then narrows \hsize to one
% column and installs \Gcoloutput: it fills the left column to the column height,
% saves it, fills the right column, and ships a page carrying the stashed
% material above the two columns. \Gfin flushes the last column.
\newbox\Gsbox \newbox\Glbox
\def\Glr{L}
\def\Gcoloutput{%
  \if L\Glr
    \global\setbox\Glbox=\box255 \gdef\Glr{R}%
  \else
    \shipout\vbox{\Gmakeheadline
      \vbox to\Gpageheight{\boxmaxdepth=\maxdimen
        \box\Gsbox\vss
        \hbox to\Gpagewidth{\box\Glbox\hfil\box255}}}%
    \global\advance\pageno by1
    \global\setbox\Gsbox=\vbox{}\global\vsize=\Gpageheight \gdef\Glr{L}%
  \fi}
\def\Ginx{\ifGinx
  \par
  \vskip 9pt plus 2pt           % gap between the Index heading/prose and entries
  \output={\global\setbox\Gsbox=\box255}\eject % stash heading+prose, hold page
  \setbox\Gsbox=\vbox{\unvbox\Gsbox}%
  \global\vsize=\Gpageheight
  \global\advance\vsize by-\ht\Gsbox \global\advance\vsize by-\dp\Gsbox
  \global\hsize=\Gcolwidth
  \global\output={\Gcoloutput}\gdef\Glr{L}%
  \begingroup
  \baselineskip=12pt \parindent=0pt \parskip=1pt plus .5pt \rm
  \rightskip=0pt plus 2.5em \tolerance=10000 \hyphenpenalty=10000
  \fi}
\def\GII#1#2{\ifGinx\noindent\hangindent2em #1:\kern.5em #2.\par\fi} % index entry
\def\GIR#1{{\rm #1}}                          % @^...@>  roman index entry
\def\GIT#1{{\tentex #1}}                      % @....@>  typewriter index entry
\def\9#1{{\rm #1}}                            % default for @:...@> entries
\def\GIC#1{\9{#1}}

% ---- list of section names (a `Names of the Sections' page) -----------------
% \Gfin ends the index (\Gendindex) and starts the section-name list
% (\Gbeginsecs); the two halves are split out so \nosecs can end the index and
% stop, and each is guarded by its own flag so the groups stay balanced.
\def\Gendindex{\par\vfil\eject  % seal/ship the current index column
  \if L\Glr\else\null\vfil\eject\fi % a left column is pending: ship its page
  \endgroup                     % close the index paragraph group (from \Ginx)
  \global\output={\plainoutput}% back to ordinary single-column pages
  \global\hsize=\Gpagewidth \global\vsize=\Gpageheight}
\def\Gbeginsecs{\vfil\eject
  \global\def\Gsechead{}% no section number in the header here
  \global\def\Ggrouphead{\Gnamesheading}%
  \begingroup
  \parindent=0pt \parskip=2pt plus 1pt \rightskip=0pt plus 2em \rm}
\def\Gfin{\ifGinx\Gendindex\fi \ifGsecs\Gbeginsecs\fi}
\def\GNS#1#2#3{\ifGsecs\noindent\hangindent2em \Gangle{#1}{#2}%
  \ifx\Gempty#3\else\quad{\Gsmallfont#3}\fi\par\fi}

% ---- contents page: centered title, Section/Page columns, dot leaders --------
% \GTline{depth}{secno}{title}{page}: one contents entry. The title travels in
% the toc file itself (written unexpanded by \GN), so the file is self-contained
% and \readcontents can gather several pieces' contents in one run.
\def\GTline#1#2#3#4{\line{\Gconsetup{#1}%
  #3\ %
  \rm\leaders\hbox to .5em{\hss.\hss}\hfil\ \Gs{#2}\hbox to 3em{\hss#4}}}
% A contents entry's leading material by group depth: a top-level @** section
% (depth -1) sets its title in bold, as cweave does; @* (0) is flush, and deeper
% groups (@*1, @*2, ...) are indented. The \bf is reset to \rm before the page
% number back in \GTline.
\def\Gconsetup#1{\ifnum#1<0 \bf \else
  \ifcase#1 \or\hskip2em\or\hskip4em\or\hskip6em\else\hskip8em\fi \fi}
\def\Gcon{\ifGsecs\par\endgroup\vfil\eject\fi % finish the section names
  \ifGtocopen\immediate\closeout\Gtoc\fi
  \ifGcon\ifGtocopen
    \titletrue \pageno=\contentspagenumber
